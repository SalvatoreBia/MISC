38static size_t _misc_str_hash_fn(const void *key)
39{
40 const char *str = *(const char**)key;
41 size_t hash = FNV_OFFSET_BASIS;
42
43 while (*str)
44 {
45 hash ^= (unsigned char)(*str++);
46 hash *= FNV_PRIME;
47 }

49 return hash;

305int misc_htable_put(misc_htable ht, const void *key, const void *value)
306{
307 if (ht == NULL || key == NULL || value == NULL) return 0;
308
309 double load_factor = (double)ht->size / (double)ht->capacity;
310 if (load_factor >= LOAD_FACTOR_THRESHOLD)
311 {
312 if (!_misc_htable_rehash(ht))
313 {
314 return 0;
315 }
316 }
317
318 size_t hash = ht->hash_func(key);
319 size_t bucket_idx = hash % ht->capacity;
320
321 misc_list *bucket = (misc_list*)misc_vector_get(ht->buckets, bucket_idx);
322
323 if (*bucket == NULL)
324 {
325 *bucket = misc_list_create(sizeof(misc_htable_entry));
326 if (*bucket == NULL) return 0;
327 }
328
329 size_t list_size = misc_list_size(*bucket);
330 for (size_t i = 0; i < list_size; i++)
331 {
332 misc_htable_entry *entry = (misc_htable_entry*)misc_list_get(*bucket, i);
333 if (ht->key_cmp(entry->key, key) == 0)
334 {
335 memcpy(entry->value, value, ht->value_size);
336 return 1;
337 }
338 }
339
340 misc_htable_entry new_entry;
341 new_entry.key = malloc(ht->key_size);
342 new_entry.value = malloc(ht->value_size);
343
344 if (new_entry.key == NULL || new_entry.value == NULL)
345 {
346 if (new_entry.key != NULL) free(new_entry.key);
347 if (new_entry.value != NULL) free(new_entry.value);
348 return 0;
349 }
350
351 memcpy(new_entry.key, key, ht->key_size);
352 memcpy(new_entry.value, value, ht->value_size);
353
354 if (!misc_list_pushback(*bucket, &new_entry))
355 {

357 free(new_entry.value);
358 return 0;
359 }
